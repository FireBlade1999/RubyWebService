require 'rake' require 'rake/clean' require 'rdoc/task' require 'rake/testtask' require 'rspec/core/rake_task'

Rake::RDocTask.new do |rdoc|

files = ['README*', 'LICENCE', 'lib/*']
rdoc.rdoc_files.add(files)
rdoc.main = "README.md" # page to start on
rdoc.title = "RubyWebService Docs"
rdoc.rdoc_dir = 'doc/rdoc' # rdoc output folder
rdoc.options << '--line-numbers'

end

RSpec::Core::RakeTask.new do |spec|

spec.pattern = 'spec/erector/*_spec.rb'
spec.rspec_opts = [Dir["lib"].to_a.join(':')]

end

Rake::TestTask.new do |t|

t.libs << "tests"
t.test_files = FileList['test/*.rb']
t.verbose = true

end
